# 1. Add Bitnami repository
helm repo add bitnami https://charts.bitnami.com/bitnami
helm repo update

# 3. Install Thanos with values file
helm upgrade --install thanos bitnami/thanos \
  --namespace opentelemetry \
  --values base/prometheus/thanos-values.yaml

oc adm policy add-scc-to-user privileged -z thanos-compactor
oc adm policy add-scc-to-user privileged -z thanos-query
oc adm policy add-scc-to-user privileged -z thanos-query-frontend
oc adm policy add-scc-to-user privileged -z thanos-storegateway
oc adm policy add-scc-to-user privileged -z thanos-receive
